#include<bits/stdc++.h> 
using namespace std;
 bool isPrime(int n)
{
    if (n <= 1)
        return false;
 
    
    for (int i = 2; i <= sqrt(n); i++)
        if (n % i == 0)
            return false; 
 // Check from 2 to square root of n
    return true;
}
